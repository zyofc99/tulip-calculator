if (!res.ok) throw new Error('Frankfurter HTTP ' + res.status); const data = await res.json(); if (!data.rates) throw new Error('Frankfurter 無 rates'); return { usdTwd: data.rates['TWD'], usdJpy: data.rates['JPY'], date: data.date }; } // ✅ API 2：OpenER async function fetchFromOpenER() { const res = await fetchWithTimeout( 'https://open.er-api.com/v6/latest/USD', { cache: 'no-store' }, 8000 ); if (!res.ok) throw new Error('OpenER HTTP ' + res.status); const date = `${twDate.getFullYear()}/${String(twDate.getMonth()+1).padStart(2,'0')}/${String(twDate.getDate()).padStart(2,'0')}`; return { usdTwd: data.rates['TWD'], usdJpy: data.rates['JPY'], date }; } // ✅ API 3：ExchangeRate-API async function fetchFromExchangeRate() { const res = await fetchWithTimeout( 'https://api.exchangerate-api.com/v4/latest/USD', { cache: 'no-store' }, 8000 ); if (!res.ok) throw new Error('ExchangeRate HTTP ' + res.status); return { usdTwd, usdJpy, date }; } // ✅ API 5：Fawaz Ahmed 備援鏡像 async function fetchFromFawazMirror() { const today = new Date(); const dateStr = `${today.getFullYear()}-${String(today.getMonth()+1).padStart(2,'0')}-${String(today.getDate()).padStart(2,'0')}`;
